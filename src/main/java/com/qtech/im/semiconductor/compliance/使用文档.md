# ComplianceChecker 使用说明文档

## 1. 概述

[ComplianceChecker](file://E:\dossier\others\im-common\src\main\java\com\qtech\im\semiconductor\compliance\ComplianceChecker.java#L32-L1538) 是一个专业的合规性检查工具类，专门为半导体封测行业的合规性管理设计。该工具解决了行业标准符合性检查困难、质量体系审核数据不完整、合规性报告生成复杂和不符合项跟踪管理困难等问题。

## 2. 核心功能

### 2.1 标准符合性检查
- 支持多种国际标准（ISO 9001、ISO 14001、IATF 16949等）
- 可配置的检查规则和评估标准
- 缓存机制提高检查效率

### 2.2 质量体系审核
- 完整的审核流程管理
- 多优先级审核项处理
- 自动生成审核报告和发现项

### 2.3 合规性报告生成
- 多维度合规性指标统计
- 自动识别主要问题
- 生成改进建议

### 2.4 不符合项跟踪
- 不符合项生命周期管理
- 趋势分析和重复问题识别
- 纠正措施有效性评估

// 标准符合性检查示例
public class StandardComplianceExample {
    public void checkStandardCompliance() {
        System.out.println("开始标准符合性检查示例...");
        
        // 1. 创建标准对象
        ComplianceChecker.Standard standard = 
            new ComplianceChecker.Standard(ComplianceChecker.StandardType.ISO9001, "2015");
        standard.setDescription("质量管理体系要求");
        
        System.out.println("1. 创建标准对象:");
        System.out.println("  标准类型: " + standard.getStandardType());
        System.out.println("  标准版本: " + standard.getVersion());
        System.out.println("  标准描述: " + standard.getDescription());
        
        // 2. 创建审核数据
        ComplianceChecker.AuditData auditData = new ComplianceChecker.AuditData(" quality_audit_2025");
        auditData.setData("document_control", true);
        auditData.setData("management_review", true);
        auditData.setData("internal_audit", true);
        auditData.setData("corrective_action", false); // 不符合项
        
        System.out.println("\n2. 创建审核数据:");
        System.out.println("  数据源: " + auditData.getSource());
        System.out.println("  数据项数: " + auditData.getData().size());
        
        // 3. 执行标准符合性检查
        ComplianceChecker.ComplianceResult result = 
            ComplianceChecker.checkStandardCompliance(standard, auditData);
        
        if (result != null) {
            System.out.println("\n3. 执行标准符合性检查:");
            System.out.println("  检查状态: " + result.getStatus());
            System.out.println("  检查消息: " + result.getMessage());
            System.out.println("  检查时间: " + new Date(result.getTimestamp()));
            
            Object details = result.getDetails();
            if (details != null) {
                System.out.println("  详细信息: " + details.toString());
            }
        } else {
            System.err.println("\n3. 标准符合性检查失败");
        }
        
        // 4. 使用配置执行检查
        System.out.println("\n4. 使用配置执行检查:");
        ComplianceChecker.ComplianceConfig config = new ComplianceChecker.ComplianceConfig()
                .setCheckTimeout(30000)
                .setMaxRetries(3)
                .setRetryDelay(1000)
                .setEnableCache(true);
        
        ComplianceChecker.ComplianceResult configResult = 
            ComplianceChecker.checkStandardCompliance(standard, auditData, config);
        
        if (configResult != null) {
            System.out.println("  配置检查状态: " + configResult.getStatus());
        }
        
        // 5. 异步执行检查
        System.out.println("\n5. 异步执行检查:");
        CompletableFuture<ComplianceChecker.ComplianceResult> futureResult = 
            ComplianceChecker.checkStandardComplianceAsync(standard, auditData, config);
        
        try {
            ComplianceChecker.ComplianceResult asyncResult = futureResult.get(10, TimeUnit.SECONDS);
            System.out.println("  异步检查完成，状态: " + asyncResult.getStatus());
        } catch (Exception e) {
            System.err.println("  异步检查超时或异常: " + e.getMessage());
        }
        
        // 6. 批量标准检查
        System.out.println("\n6. 批量标准检查:");
        List<ComplianceChecker.Standard> standards = Arrays.asList(
            new ComplianceChecker.Standard(ComplianceChecker.StandardType.ISO9001, "2015"),
            new ComplianceChecker.Standard(ComplianceChecker.StandardType.ISO14001, "2015"),
            new ComplianceChecker.Standard(ComplianceChecker.StandardType.IATF16949, "2016")
        );
        
        List<ComplianceChecker.ComplianceResult> batchResults = 
            ComplianceChecker.checkBatchCompliance(standards, auditData, config);
        
        System.out.println("  批量检查完成，共检查 " + batchResults.size() + " 个标准");
        for (int i = 0; i < batchResults.size(); i++) {
            ComplianceChecker.ComplianceResult batchResult = batchResults.get(i);
            System.out.println("    " + standards.get(i).getStandardType() + ": " + batchResult.getStatus());
        }
    }
}

            if (reportScope != null) {
                System.out.println("  报告范围: " + reportScope.getScopeName());
                System.out.println("  范围类型: " + reportScope.getScopeType());

            // 显示指标信息

            if (metrics != null) {
                System.out.println("  总审核数: " + metrics.getTotalAudits());
                System.out.println("  通过审核数: " + metrics.getPassedAudits());
                System.out.println("  失败审核数: " + metrics.getFailedAudits());
                System.out.println("  合规率: " + String.format("%.2f", metrics.getComplianceRate() * 100) + "%");
            }
            
            // 显示主要问题
            List<ComplianceChecker.ComplianceIssue> majorIssues = complianceReport.getMajorIssues();
            System.out.println("  主要问题数: " + majorIssues.size());
            if (!majorIssues.isEmpty()) {
                System.out.println("  主要问题列表:");
                for (ComplianceChecker.ComplianceIssue issue : majorIssues) {
                    System.out.println("    类型: " + issue.getIssueType());
                    System.out.println("    出现次数: " + issue.getOccurrenceCount());
                    System.out.println("    描述: " + issue.getDescription());
                }
            }
            
            // 显示改进建议
            List<ComplianceChecker.ImprovementRecommendation> recommendations = 
                complianceReport.getRecommendations();
            System.out.println("  改进建议数: " + recommendations.size());
            if (!recommendations.isEmpty()) {
                System.out.println("  改进建议列表:");
                for (ComplianceChecker.ImprovementRecommendation recommendation : recommendations) {
                    System.out.println("    描述: " + recommendation.getDescription());
                    System.out.println("    优先级: " + recommendation.getPriority());
                    System.out.println("    理由: " + recommendation.getRationale());
                }
            }
            
            // 显示审核报告
            List<ComplianceChecker.AuditReport> auditReports = complianceReport.getAuditReports();
            System.out.println("  包含审核报告数: " + auditReports.size());
        } else {
            System.err.println("\n2. 合规性报告生成失败");
        }

        System.out.println("\n3. 使用配置生成报告:");
        ComplianceChecker.ReportConfig reportConfig = new ComplianceChecker.ReportConfig()
                .setTemplate("detailed")
                .setIncludeCharts(true)
                .setIncludeRecommendations(true)
                .setMaxIssuesToShow(20);
        
        ComplianceChecker.ComplianceReport configReport = 
            ComplianceChecker.generateComplianceReport(scope, reportConfig);
        
        if (configReport != null) {
            System.out.println("  配置报告生成成功");
            System.out.println("  主要问题数: " + configReport.getMajorIssues().size());
            System.out.println("  改进建议数: " + configReport.getRecommendations().size());
        }
    }

public class ComplianceCheckerErrorHandler {
    public void handleComplianceErrors() {
        System.out.println("合规性检查器错误处理示例");
        
        try {
            // 创建无效的标准对象
            ComplianceChecker.Standard invalidStandard = 
                new ComplianceChecker.Standard(null, null);
            
            // 创建审核数据
            ComplianceChecker.AuditData auditData = new ComplianceChecker.AuditData("testing");
            
            // 尝试执行检查（预期会处理错误）
            ComplianceChecker.ComplianceResult result =

            if (result != null && result.getStatus() == ComplianceChecker.ComplianceStatus.FAILURE) {
                System.err.println("标准符合性检查失败: " + result.getMessage());
                handleComplianceError(result);
            }
            
        } catch (Exception e) {
            System.err.println("合规性检查器使用过程中发生异常: " + e.getMessage());
            e.printStackTrace();
        }
    }
    
    private void handleComplianceError(ComplianceChecker.ComplianceResult result) {
        switch (result.getStatus()) {
            case FAILURE:
                System.err.println("  检查执行一般性失败");
                break;
            case NON_COMPLIANT:
                System.err.println("  标准不符合");
                break;
            case PENDING:
                System.err.println("  检查待处理");
                break;
            default:
                System.err.println("  未知错误状态: " + result.getStatus());
        }
        
        // 处理详细错误信息
        Object details = result.getDetails();
        if (details != null) {
            System.err.println("  详细错误信息: " + details.toString());
        }
    }
    
    public boolean safeComplianceCheck(ComplianceChecker.Standard standard, 
                                     ComplianceChecker.AuditData data) {
        if (standard == null || data == null) {
            System.err.println("标准或审核数据不能为空");
            return false;
        }
        
        ComplianceChecker.ComplianceConfig config = new ComplianceChecker.ComplianceConfig()
                .setMaxRetries(3)
                .setRetryDelay(1000);
        
        int maxAttempts = 3;
        int attemptCount = 0;
        
        while (attemptCount < maxAttempts) {
            try {
                ComplianceChecker.ComplianceResult result = 
                    ComplianceChecker.checkStandardCompliance(standard, data, config);
                
                if (result != null && result.getStatus() != ComplianceChecker.ComplianceStatus.FAILURE) {
                    System.out.println("合规性检查成功: " + result.getStatus());
                    return true;
                } else {
                    System.err.println("第" + (attemptCount + 1) + "次检查失败: " + 
                                     (result != null ? result.getMessage() : "null"));
                }
                
            } catch (Exception e) {
                System.err.println("第" + (attemptCount + 1) + "次检查异常: " + e.getMessage());
            }
            
            attemptCount++;
            
            // 重试前等待
            if (attemptCount < maxAttempts) {
                try {
                    Thread.sleep(1000 * attemptCount); // 递增延迟
                } catch (InterruptedException e) {
                    Thread.currentThread().interrupt();
                    break;
                }
            }
        }
        
        System.err.println("合规性检查最终失败，已重试 " + maxAttempts + " 次");
        return false;
    }
}
